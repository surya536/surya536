import java.util.Scanner;

public class ques19c {
	public static void main(String args[])   
	{   
	int i, j, r,k=1; 
	Scanner sc=new Scanner(System.in);
	System.out.println("enter rows");
	r=sc.nextInt();
	for(i=1; i<=r; i++)   
	{   
	for(j=1; j<=i; j++)   
	{     
	System.out.print(k++ + " ");   
	}   
	System.out.println();   
	}   
	}   


}
