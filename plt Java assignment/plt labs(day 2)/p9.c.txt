	public static void main(String args[])
	{
		int a=1,b=5,c=8,d;
		Scanner sc=new Scanner(System.in);
		System.out.print("enter the limit");
		int n=sc.nextInt();
		System.out.print(a+" ");
		System.out.print(b+" ");

		for(int i=0;i<n;i++)
		{
			d=a+b+c;
			System.out.print(d+" ");
			a=b;
			b=c;
			c=d;
			
		}
	}

}
